############# PARAMETERS ################################################
n <- 5
template <- 1
phi <- pi/5
e <- 0.02
eta <- pi/50
clean.y <- c(template*cos(phi),template*sin(phi))

############## FUNCTIONS ################################################
## ~~ Generative model without errors ~~ ##
generate=function(my.n,witherrors=FALSE,my.e,my.eta=0){
  noisy.y <- mvrnorm(my.n,clean.y,my.e*diag(2))
  clean.phi <- atan(noisy.y[,2]/noisy.y[,1]) #the clean angle for registration
  Data <-data.frame(clean.phi,I(noisy.y))
  if (witherrors){
    delta.phi <- rnorm(my.n,mean=0,sd=my.eta)
    dirty.phi <- clean.phi + delta.phi
    Data <- cbind(Data,data.frame(dirty.phi))
  }
  return (Data)
}
## ~~ Learning algorithm ~~ ##
learn=function(my.Data,witherrors=FALSE){
  my.n <-nrow(my.Data)
  my.noisy.y <-my.Data$noisy.y
  Rotinv <- array(0,c(my.n,2,2))
  z <- matrix(0,ncol=2,nrow=my.n) #points after registration
  for (i in 1:my.n){
    if (witherrors)  {
      my.dirty.phi <-my.Data$dirty.phi
      Rotinv[i,,]=matrix(c(cos(my.dirty.phi[i]),sin(my.dirty.phi[i]),-sin(my.dirty.phi[i]),cos(my.dirty.phi[i])),nrow=2,ncol=2,byrow=TRUE)  
    } else {
      my.clean.phi <- my.Data$clean.phi
      Rotinv[i,,]=matrix(c(cos(my.clean.phi[i]),sin(my.clean.phi[i]),-sin(my.clean.phi[i]),cos(my.clean.phi[i])),nrow=2,ncol=2,byrow=TRUE)
    }
    z[i,]=Rotinv[i,,]%*%my.noisy.y[i,]
  }
  return(z)
}
## ~~ Plot 2D picture ~~ ##
my.plot <-function(my.Data,my.z,witherrors=FALSE,my.e,my.eta=0){
  my.n <-nrow(my.Data)
  qmean <-c(mean(my.z[,1]),mean(my.z[,2]))
  my.noisy.y <-my.Data$noisy.y
  par(mfrow=c(1,1),mar=c(1,1,2,1))
  plot(my.noisy.y[,1],my.noisy.y[,2],xlim=c(-1,2),ylim=c(-1,2),col="darkgrey",pch=19,axis=FALSE, xaxt="n",yaxt="n")
  if (witherrors){
    my.title=paste('Frechet mean for n=',my.n,"noise e=",e,"and meas. error eta=",my.eta)
  } else {
    my.title=paste('Frechet mean for n=',my.n,"noise e=",e, "and no meas. error")
  }
  title(main=my.title,cex.main=0.8)
  axis(side=1,pos=c(0,0),outer=FALSE, cex.axis=0.7,at=c(-1,0,1,2),tck=0.01,xpd = NA)
  axis(side=2,pos=c(0,0),outer=FALSE, cex.axis=0.7,at=c(-1,0,1,2),las=2,tck=0.01,xpd = TRUE)
  # Plot real mean in green
  points(clean.y[1],clean.y[2], col="green",lwd=6, type="p",pch=19)
  segments(x0 = 0, x1 = clean.y[1],y0 = 0, y1 = clean.y[2], col = "green",lwd=4)
  # Registered points and qFrechet mean
  points(my.z[,1],my.z[,2], col="black", type="p",pch=19)
  points(qmean[1],qmean[2],col="blue",lwd=6,type="p",pch=19)
  segments(x0 = 0, x1 =qmean[1],y0 = 0, y1 = qmean[2], col = "blue",lwd=4)
}
## ~~ Plot Histogram ~~ ##
my.hist <- function(my.templ, my.unctcy,my.qmean,witherrors=FALSE){
  par(mfrow=c(1,1),mar=c(3,3,3,0))
  my.Nexp <-nrow(my.qmean)
  my.norm <-rep(0,my.Nexp)
  for (i in 1:my.Nexp)
  {
    my.norm[i]=sqrt(sum(my.qmean[i,]^2))
  }
  title=paste("Hist of qFrechet mean, Nexp=", Nexp,", npts=",npts,"witherrors=",witherrors)
  hist(my.norm,main=title,breaks=20, cex=0.7, cex.main=0.8)
  abline(v=my.templ, col = "green", lwd=4)
  abline(v=my.templ+my.unctcy, col = "darkgreen", lwd=4)
  abline(v=mean(my.norm), col = "blue", lwd=4)
  legend('topright', c('template','template + unconstcy.err','qFrechet mean'), lty=1, lwd=4,col=c('green','darkgreen','blue'), bty='n', cex=.75)
}
